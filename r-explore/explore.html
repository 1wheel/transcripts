Use dplyr mutate to add a time and length variable to each call

calls <- calls %>%
  mutate(length = nchar(text)) %>%
  mutate(dateNum = as.numeric(year) + as.numeric(substr(x = quarter, start = 2, stop = 2))/4 - .25) %>%
  filter(dateNum < 2015.5) %>%
  filter(length > 0)
